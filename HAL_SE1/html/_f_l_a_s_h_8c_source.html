75 unsigned int _FLASH_GetSector( void* addr )
76 {
77  unsigned int uiAddr = (unsigned int)addr;
78 
79  if ( uiAddr >= 0x00000000 && uiAddr <= 0x0000FFFF )
80  return uiAddr >> 0xC;
81 
82  else if ( uiAddr >= 0x00010000 && uiAddr <= 0x0007FFFF )
83  return ((uiAddr - 0x00010000) >> 0xF) | 0x10;
84 
85  //Address does not belong to any sector
86  return -1;
87 }
