39 void WAIT_ChronosInit()
40 {
41  LPC_SC->PCONP |= TIMER0_BIT; //Enable Timer 0
42  LPC_SC->PCLKSEL0 &= ~(0x3 << 2); //1/4 CCLK
43  LPC_TIM0->PR = (SystemCoreClock/4)/1000000 - 1;
44  LPC_TIM0->TCR = COUNTER_OFF | RESET_ON; //Counter disabled and Reset Enabled
45  LPC_TIM0->IR = CLEAR_MR0I; //Reset the interrupt for match channel 0
46  NVIC_EnableIRQ(TIMER0_IRQn);
47 }

49 int WAIT_ChronoUs(uint32_t waitUs)
50 {

52  LPC_TIM0->MR0 = waitUs;
53  LPC_TIM0->MCR = MR0I | MR0S; //Enable interrupt on match 0 and disable counter on match 0
54  LPC_TIM0->TCR = COUNTER_ON | RESET_OFF; //Counter enabled and Reset disabled
55  while(!__CHEGUEI)
56  {
57  __WFI();
58  }
59 
60  return 0;
