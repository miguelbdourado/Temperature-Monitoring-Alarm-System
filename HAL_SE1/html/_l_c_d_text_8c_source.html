10  * P0[23..26] -> DB4-DB7 -> 11-14
11  * P1[30..31] -> RS/E -> 4/6

13 void LCDText_Init()
14 {
15  //Init chrono
16  WAIT_ChronosInit();
17 
18  //Set GPIO function for pins P0[23..26] and P1[30..31]
19  LPC_PINCON->PINSEL1 &= ~(0x3FC000);
20  LPC_PINCON->PINSEL3 &= ~(0xF0000000);
21 
22  //Set output mode for pins P0[23..26] and P1[30..31]
23  LPC_GPIO0->FIODIR |= (0x7800000);
24  LPC_GPIO1->FIODIR |= (0xC0000000);
25 
26  __row = 0;
27  __column = 0;
28 
29  //Wait more than 15ms
30  WAIT_ChronoUs(20 * 1000); //20 ms
31 
32  //Step 1
33  __LCD_Write5Bits(0, 0x3);
34 
35  //Wait more than 5ms
36  WAIT_ChronoUs(10 * 1000); //10ms
37 
38  //Step 2
39  __LCD_Write5Bits(0, 0x3);
40 
41  //Wait more than 100 nanoseconds
42  WAIT_ChronoUs(1);
43 
44  //Step 3
45  __LCD_Write5Bits(0, 0x3);
46 
47  //Step 4
48  __LCD_Write5Bits(0, 0x2);
49  __LCD_Write9Bits(0, 0x28);//2 lines and 5x8 dots
50  __LCD_Write9Bits(0, 0x8);
51  __LCD_Write9Bits(0, 0x1); //Clear display
52  __LCD_Write9Bits(0, 0x6); //Increments cursor on writes
53 
54  WAIT_ChronoUs(10 * 1000); //10ms
55  __LCD_Write9Bits(0, 0xF); //Turn on display, turn on blinking cursor
56 }

58 void LCDText_SetCursor(int row, int column) {
59  int new_row = 0, new_column = 0;
60 
61  //Clamp row
62  if(row < 0) new_row = 0;
63  else if(row >= MAX_ROW) new_row = MAX_ROW - 1;
64  else new_row = row;
65 
66  //Clamp column
67  if(column < 0) new_column = 0;
68  else if(column >= MAX_COLUMN) new_column = MAX_COLUMN - 1;
69  else new_column = column;
70 
71  int new_cursor_pos = (new_row * 40) + new_column;
72  int old_cursor_pos = (__row * 40) + __column;
73 
74  int delta_cursor_pos = new_cursor_pos - old_cursor_pos;
75 
76  int8_t data = 0x10; //Shift to the left
77  if(delta_cursor_pos > 0) data |= 0x4; //Shift to the right
78 
79  for(int i = 0; i < abs(delta_cursor_pos); i++)
80  {
81  __LCD_Write9Bits(0, data);
82  WAIT_ChronoUs(1);
83  }
84 
85  __column = new_column;
86  __row = new_row;
87 }

113 void __LCD_Write5Bits(bool rs, int8_t data)
114 {
115  if(rs) LPC_GPIO1->FIOSET = (1 << RS_POS_ID);
116  else LPC_GPIO1->FIOCLR = (1 << RS_POS_ID);
117 
118  LPC_GPIO0->FIOCLR = (0xF << DATA_LOW_POS_ID);
119  LPC_GPIO0->FIOSET = ((data & 0xF) << DATA_LOW_POS_ID);
120  PULSE_ENABLE();
121 }

123 void LCDText_WriteChar(char ch)
124 {
125  __LCD_Write9Bits(1, ch);
126  __column++;
127 
128  //If we are writing to a column outside the screen range, update row and column on the LCD
129  if(__column >= MAX_COLUMN)
130  {
131  int new_row = 0, new_column = 0;
132 
133  if(__row == MAX_ROW - 1) new_row = 0;
134  else new_row = 1;
135 
136  new_column = 0;
137 
138  LCDText_SetCursor(new_row, new_column);
139  }
140 }
