49 {
50  /*
51  * P0.15 -> MODE 11 (SCK) OUTPUT
52  * P0.16 -> MODE 11 (SSEL) OUTPUT
53  * P0.17 -> MODE 11 (MISO) INPUT sdo
54  * P0.18 -> MODE 11 (MOSI) OUTPUT sdi
55  */
56  LPC_PINCON->PINSEL0 |= (0x3 << 30);
57  LPC_PINCON->PINSEL1 |= 0x3F;
58 
59  LPC_GPIO0->FIODIR |= (1 << 15); //OUTPUT
60  LPC_GPIO0->FIODIR |= (1 << 16); //OUTPUT
61  LPC_GPIO0->FIODIR &= ~(1 << 17); // INPUT
62  LPC_GPIO0->FIODIR |= (1 << 18); //OUTPUT
63 
64  LPC_GPIO0->FIOSET |= (1 << 16); //Disable SSEL
65 
66  LPC_SC->PCONP |= ENABLE_SPI;
67 
68  LPC_SPI->SPCCR = spi_GetPclk() / 4000000; //freq 25
69 
70  LPC_SPI->SPCR = (1 << CPHA_MASK) | (1 << CPOL_MASK) | (1 << MSTR_MASK);
